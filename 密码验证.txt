    private static boolean hasKinds(String s) {
        char[] chars=s.toCharArray();
        int d=0;//Êý×Ö
        int u=0;//´óÐ´×ÖÄ¸
        int low=0;//Ð¡Ð´×ÖÄ¸
        int o=0;//ÆäËû×ÖÄ¸
        for (char c:chars ) {

            }
            else if(c>='a'&&c<='z') {
                low=1;

            }else if (c>='A'&&c<='Z') {

    private static boolean containsTwo(String s) {
        for (int i = 0; i < s.length()-2; i++) {
            String flag=s.substring(i,i+3);
            if (s.substring(i+1).contains(flag)) {
                return false;
            }
        }
        return true;
    }
